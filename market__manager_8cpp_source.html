629 void MarketManager::Match(OrderBook* order_book_ptr)
630 {
631  // Check the arbitrage bid/ask prices
632  while ((order_book_ptr->_best_bid != nullptr) &&
633  (order_book_ptr->_best_ask != nullptr) &&
634  (order_book_ptr->_best_bid->Price >= order_book_ptr->_best_ask->Price))
635  {
636  LevelNode* bid = order_book_ptr->_best_bid;
637  LevelNode* ask = order_book_ptr->_best_ask;
638 
639  // Execute crossed orders
640  while (!bid->OrderList.empty() && !ask->OrderList.empty())
641  {
642  // Find the order to execute and the order to reduce
643  OrderNode* bid_order_ptr = bid->OrderList.front();
644  OrderNode* ask_order_ptr = ask->OrderList.front();
645 
646  // Special case for 'All-Or-None' orders
647  if (bid_order_ptr->IsAON() || ask_order_ptr->IsAON())
648  {
649  // Calculate the 'All-Or-None' orders chain
650  uint64_t chain = CalculateAllOrNoneChain(bid_order_ptr, ask_order_ptr);
651 
652  // Matching is not avaliable
653  if (chain == 0)
654  return;
655 
656  // Execute bid orders
657  ExecuteAllOrNoneChain(bid_order_ptr, chain);
658 
659  // Execute ask orders
660  ExecuteAllOrNoneChain(ask_order_ptr, chain);
661  }
662  else
663  {
664  // Find the order to execute and the order to reduce
665  OrderNode* executing_order_ptr = bid_order_ptr;
666  OrderNode* reducing_order_ptr = ask_order_ptr;
667  if (executing_order_ptr->Quantity > reducing_order_ptr->Quantity)
668  std::swap(executing_order_ptr, reducing_order_ptr);
669 
670  // Get the execution quantity
671  uint64_t quantity = executing_order_ptr->Quantity;
672 
673  // Get the execution price
674  uint64_t price = executing_order_ptr->Price;
675 
676  // Call the corresponding handler
677  _market_handler.onExecuteOrder(*executing_order_ptr, price, quantity);
678 
679  // Delete the executing order from the order book
680  DeleteOrder(executing_order_ptr->Id);
681 
682  // Call the corresponding handler
683  _market_handler.onExecuteOrder(*reducing_order_ptr, price, quantity);
684 
685  // Reduce the remaining order in the order book
686  ReduceOrder(reducing_order_ptr->Id, quantity);
687  }

692 void MarketManager::MatchMarket(OrderBook* order_book_ptr, Order* order_ptr)
693 {
694  // Check if the order book is valid
695  if ((order_book_ptr == nullptr) || (order_ptr == nullptr))
696  return;
697 
698  // Calculate acceptable marker order price with optional slippage value
699  if (order_ptr->IsBuy())
700  {
701  // Check if there is nothing to buy
702  if (order_book_ptr->best_ask() == nullptr)
703  return;
704 
705  order_ptr->Price = order_book_ptr->best_ask()->Price;
706  if (order_ptr->Price > (std::numeric_limits<uint64_t>::max() - order_ptr->Slippage))
707  order_ptr->Price = std::numeric_limits<uint64_t>::max();
708  else
709  order_ptr->Price += order_ptr->Slippage;
710  }
711  else
712  {
713  // Check if there is nothing to sell
714  if (order_book_ptr->best_bid() == nullptr)
715  return;
716 
717  order_ptr->Price = order_book_ptr->best_bid()->Price;
718  if (order_ptr->Price < (std::numeric_limits<uint64_t>::min() + order_ptr->Slippage))
719  order_ptr->Price = std::numeric_limits<uint64_t>::min();
720  else
721  order_ptr->Price -= order_ptr->Slippage;
722  }
723 
724  // Match the market order
725  MatchOrder(order_book_ptr, order_ptr);

738 void MarketManager::MatchOrder(OrderBook* order_book_ptr, Order* order_ptr)
739 {
740  // Start the matching from the top of the book
741  LevelNode* level;
742  while ((level = order_ptr->IsBuy() ? order_book_ptr->_best_ask : order_book_ptr->_best_bid) != nullptr)
743  {
744  // Check the arbitrage bid/ask prices
745  bool arbitrage = order_ptr->IsBuy() ? (order_ptr->Price >= level->Price) : (order_ptr->Price <= level->Price);
746  if (!arbitrage)
747  return;
748 
749  // Execute crossed orders
750  while (!level->OrderList.empty())
751  {
752  // Find the order to execute
753  OrderNode* executing_order_ptr = level->OrderList.front();
754 
755  // Get the execution quantity
756  uint64_t quantity = std::min(executing_order_ptr->Quantity, order_ptr->Quantity);
757 
758  // Special case for 'All-Or-None' orders
759  if (executing_order_ptr->IsAON() && order_ptr->IsAON())
760  {
761  // In order to match two 'All-Or-None' orders their quantities must be equal
762  if (executing_order_ptr->Quantity != order_ptr->Quantity)
763  return;
764  }
765  else if (executing_order_ptr->IsAON() && (executing_order_ptr->Quantity > order_ptr->Quantity))
766  return;
767  else if (order_ptr->IsAON() && (order_ptr->Quantity > executing_order_ptr->Quantity))
768  return;
769 
770  // Get the execution price
771  uint64_t price = executing_order_ptr->Price;
772 
773  // Call the corresponding handler
774  _market_handler.onExecuteOrder(*executing_order_ptr, price, quantity);
775 
776  // Reduce the executing order in the order book
777  ReduceOrder(executing_order_ptr->Id, quantity);
778 
779  // Call the corresponding handler
780  _market_handler.onExecuteOrder(*order_ptr, price, quantity);
781 
782  // Reduce quantity to execute
783  order_ptr->Quantity -= quantity;
784  if (order_ptr->Quantity == 0)
785  return;
786  }
787  }
788 }

790 uint64_t MarketManager::CalculateAllOrNoneChain(OrderNode* bid_order_ptr, OrderNode* ask_order_ptr)
791 {
792  uint64_t required;
793  uint64_t available;
794  OrderNode* longest;
795  OrderNode* shortest;
796 
797  // Find the initial longest order chain
798  if (bid_order_ptr->IsAON() && ask_order_ptr->IsAON())
799  {
800  // Choose the longest 'All-Or-None' order
801  if (bid_order_ptr->Quantity > ask_order_ptr->Quantity)
802  {
803  required = bid_order_ptr->Quantity;
804  available = 0;
805  longest = bid_order_ptr;
806  shortest = ask_order_ptr;
807  }
808  else
809  {
810  required = ask_order_ptr->Quantity;
811  available = 0;
812  longest = ask_order_ptr;
813  shortest = bid_order_ptr;
814  }
815  }
816  else if (bid_order_ptr->IsAON())
817  {
818  required = bid_order_ptr->Quantity;
819  available = 0;
820  longest = bid_order_ptr;
821  shortest = ask_order_ptr;
822  }
823  else
824  {
825  required = ask_order_ptr->Quantity;
826  available = 0;
827  longest = ask_order_ptr;
828  shortest = bid_order_ptr;
829  }
830 
831  // Find volume to match
832  while (shortest != nullptr)
833  {
834  uint64_t need = required - available;
835  uint64_t volume = shortest->IsAON() ? shortest->Quantity : std::min(shortest->Quantity, need);
836  available += volume;
837 
838  // Matching is possible, return the chain size
839  if (required == available)
840  return required;
841 
842  // Swap longest and shortest chains
843  if (required < available)
844  {
845  OrderNode* next = longest->next;
846  longest = shortest;
847  shortest = next;
848  std::swap(required, available);
849  continue;
850  }
851 
852  // Take the next order
853  shortest = shortest->next;
854  }
855 
856  // Matching is not available
857  return 0;
858 }

860 void MarketManager::ExecuteAllOrNoneChain(OrderNode* order_ptr, uint64_t chain)
861 {
862  // Execute all orders in chain
863  while (chain > 0)
864  {
865  OrderNode* executing_order_ptr = order_ptr;
866  order_ptr = executing_order_ptr->next;
867 
868  uint64_t quantity;
869 
870  // Execute order
871  if (executing_order_ptr->IsAON())
872  {
873  // Get the execution quantity
874  quantity = executing_order_ptr->Quantity;
875 
876  // Call the corresponding handler
877  _market_handler.onExecuteOrder(*executing_order_ptr, executing_order_ptr->Price, quantity);
878 
879  // Delete the executing order from the order book
880  DeleteOrder(executing_order_ptr->Id);
881  }
882  else
883  {
884  // Get the execution quantity
885  quantity = std::min(executing_order_ptr->Quantity, chain);
886 
887  // Call the corresponding handler
888  _market_handler.onExecuteOrder(*executing_order_ptr, executing_order_ptr->Price, quantity);
889 
890  // Reduce the executing order in the order book
891  ReduceOrder(executing_order_ptr->Id, quantity);
892  }
893 
894  // Reduce the execution chain
895  chain -= quantity;
896  }
