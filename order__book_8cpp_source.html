99 LevelUpdate OrderBook::AddOrder(OrderNode* order_ptr)
100 {
101  // Find the price level for the order
102  LevelNode* level_ptr = order_ptr->IsBuy() ? (LevelNode*)GetBid(order_ptr->Price) : (LevelNode*)GetAsk(order_ptr->Price);
103 
104  // Create a new price level if no one found
105  UpdateType update = UpdateType::UPDATE;
106  if (level_ptr == nullptr)
107  {
108  level_ptr = AddLevel(order_ptr);
109  update = UpdateType::ADD;
110  }
111 
112  // Update the price level volume
113  level_ptr->TotalVolume += order_ptr->Quantity;
114  level_ptr->HiddenVolume += order_ptr->HiddenQuantity();
115  level_ptr->VisibleVolume += order_ptr->VisibleQuantity();
116 
117  // Link the new order to the orders list of the price level
118  level_ptr->OrderList.push_back(*order_ptr);
119  ++level_ptr->Orders;
120 
121  // Cache the price level in the given order
122  order_ptr->Level = level_ptr;
123 
124  // Price level was changed. Return top of the book modification flag.
125  return LevelUpdate(update, *order_ptr->Level, (order_ptr->Level == (order_ptr->IsBuy() ? _best_bid : _best_ask)));
126 }
